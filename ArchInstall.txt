    * para setar o horario

> fdisk -l
    * para listar os discos

> cfdisk /dev/sda
    * seleciona o disco a ser formatado e usado
    * neste caso o sda

- particoes
    * EFI             | 512MB ~ 1GB
    * swap            | 2GB
    * /               | 20 ~ 50GB
    * /HOME(opcional) | resto do espaco disponivel
    
- mount partitions
    > mkfs.ext4 /dev/sda4
        * monta a particao /home como ext4, neste caso o sda4
        
    > mkfs.ext4 /dev/sda3
        * monta a particao / como ext4, neste caso o sda3
        
    > mkswap /dev/sda2
        * monta o swap como swap, neste caso o sda2
        
    > mkfs.fat -F32 /dev/sda1
        * monta o efi como fat32
        
- mount the file system
    > mount /dev/sda3 /mnt
        * monta a particao / no /mnt
    
    > mkdir /mnt/home
        * cria o diretorio home
    
    > mkdir /mnt/boot
        * cria o diretorio de boot
        
    > mkdir /mnt/boot/efi
        * crio o diretorio do sistema efi
    
    > mount /dev/sda4 /mnt/home
        * monta a particao home
        
    > mount /dev/sda1 /mnt/boot/efi
        * monta a particao efi
        
    > swapon /dev/sda2
        * ativa a swap
    
    > echo "Server = http://archlinux.c3sl.ufpr.br/$repo/os/$arch" >> /etc/pacman.d/mirrorlist
	* seleciona um melhor mirror para download

 - basic intalation
    
    > pacstrap /mnt base linux linux-firmware vim base-devel
        * instala os recursos nescesarios como kernel e firmwares
    
    > genfstab -U -p /mnt >> /mnt/etc/fstab
        * cria o arquivo de configuracao das particoes
    
    > arch-chroot /mnt
        * loga no sistema recem criado
        
    > echo KEYMAP=br-abnt2 >> /etc/vconsole.conf
        * opcinal
        * para mudar o keyboard para pt br
            
    > echo "redfox" > /etc/hostname
        * defina o nome de usuario
        
    > vim /etc/hosts
        * altere para o nome desejado
        * o mesmo que o hostname
        * escreva neste documento:
            127.0.0.1    localhost          localhost
            ::1          localhost          localhost
            127.0.1.1    redfox.localdomain    redfox 
    
    > passwd
        * para configurar a senha de root
        
    > useradd -m -g users -G wheel redfox
        * para adicionar o usuario ao login 
        * usarei o user_name redfox

    > passwd redfox
	* para configurar a senha do user redfox
    
    > pacman -S dosfstools os-prober mtools network-manager-applet networkmanager dialog sudo
        * intalar pacotes necessarios
        
    > echo "redfox ALL=(ALL) ALL" >>  /etc/sudoers
        * para adicionar o usuario a lista de sudo
                
    > mkdir /home/arch
        * para criar o /home para o user arch
        * unicamente se este nao existir
